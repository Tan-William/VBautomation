' fill as 1,2,3....and so on, on every alternate column, continues with every row.
Dim i As Integer, j As Integer, k As Integer

k = 1
For i = 1 To 10
For j = 1 To 10 Step 2
   Cells(i, j) = k
   k = k + 1
Next j
Next i
' fill as 1,3,5,7,9 on every alternate column and on every alternate row.

Dim i As Integer, j As Integer

For i = 1 To 10 Step 2
For j = 1 To 10 Step 2
   Cells(i, j) = j
Next j
Next i

 

 
' fill as 1,2,3....continuously on every alternate column and on alternate row
' shade every alternate row with GREEN

Dim i As Integer, j As Integer, k As Integer, L As Integer
L = 1
For i = 1 To 10 Step 2
For j = 1 To 10 Step 2
   Cells(i, j) = L
   L = L + 1
       For k = 1 To 9
          Cells(i, k).Interior.ColorIndex = 4
       Next k
Next j
Next i
